<?php

namespace App\Actions\Web;

use App\Actions\BaseAction;
use Psr\Http\Message\ResponseInterface;
use Psr\Http\Message\ServerRequestInterface;

class LoginAction extends BaseAction
{
    public function __invoke(ServerRequestInterface $request, ResponseInterface $response): ResponseInterface
    {
        $title = 'Sign In | Sugar Clone';
        $description = 'Sign in to your Sugar Clone account. Connect with attractive sugar babies and generous sugar daddies!';
        
        $html = $this->view->render('pages/login', [
            'title' => $title,
            'description' => $description,
        ]);
        
        $response->getBody()->write($html);
        return $response->withHeader('Content-Type', 'text/html');
    }
}

/**
 * Login Page JavaScript
 * Handles form submission, validation, and user experience
 */

document.addEventListener('DOMContentLoaded', function() {
    const loginForm = document.getElementById('login-form');
    const submitButton = document.getElementById('login-submit');
    const formErrors = document.getElementById('form-errors');
    
    if (loginForm) {
        loginForm.addEventListener('submit', handleFormSubmit);
    }
    
    // Handle form submission
    function handleFormSubmit(event) {
        event.preventDefault();
        
        // Show loading state
        setLoadingState(true);
        
        // Get form data
        const formData = new FormData(loginForm);
        const data = Object.fromEntries(formData);
        
        // Basic validation
        if (!validateForm(data)) {
            setLoadingState(false);
            return;
        }
        
        // Submit form via HTMX
        htmx.ajax('POST', '/api/v1/auth/login', {
            target: '#login-form',
            swap: 'outerHTML',
            values: data,
            beforeRequest: function() {
                setLoadingState(true);
            },
            afterRequest: function() {
                setLoadingState(false);
            }
        });

        const errors = [];
        
        if (!data.identity || data.identity.trim() === '') {
            errors.push('Email or username is required');
        }
        
        if (!data.password || data.password.trim() === '') {
            errors.push('Password is required');
        }
        
        if (errors.length > 0) {
            showErrors(errors);
            return false;
        }
        
        hideErrors();
        return true;
    }
    
    // Show error messages
    function showErrors(errors) {
        if (formErrors && errors.length > 0) {
            const errorMessage = formErrors.querySelector('.error-message');
            if (errorMessage) {
                errorMessage.textContent = errors.join(', ');
                formErrors.style.display = 'block';
            }
        }
    }
    
    // Hide error messages
    function hideErrors() {
        if (formErrors) {
            formErrors.style.display = 'none';
        }
    }
    
    // Set loading state
    function setLoadingState(loading) {
        if (submitButton) {
            if (loading) {
                submitButton.classList.add('loading');
                submitButton.disabled = true;
            } else {
                submitButton.classList.remove('loading');
                submitButton.disabled = false;
            }
        }
    }
    
    // Clear form on successful submission
    function clearForm() {
        if (loginForm) {
            loginForm.reset();
            hideErrors();
        }
    }
    
    // Handle successful login (redirect)
    function handleSuccessfulLogin(response) {
        // Check if response indicates success
        if (response.success) {
            // Redirect to dashboard or home page
            window.location.href = response.redirectUrl || '/';
        }
    }
    
    // Add input event listeners for real-time validation
    const inputs = loginForm?.querySelectorAll('.form-input');
    if (inputs) {
        inputs.forEach(input => {
            input.addEventListener('input', function() {
                // Clear errors when user starts typing
                if (formErrors && formErrors.style.display !== 'none') {
                    hideErrors();
                }
            });
        });

    const formErrors = document.getElementById('form-errors');
    
    if (loginForm) {
        loginForm.addEventListener('submit', handleFormSubmit);
    }
    
    // Handle form submission
    function handleFormSubmit(event) {
        event.preventDefault();
        
        // Show loading state
        setLoadingState(true);
        
        // Get form data
        const formData = new FormData(loginForm);
        const data = Object.fromEntries(formData);
        
        // Basic validation
        if (!validateForm(data)) {
            setLoadingState(false);
            return;
        }
        
        // Submit form via HTMX
        htmx.ajax('POST', '/api/v1/auth/login', {
            target: '#login-form',
            swap: 'outerHTML',
            values: data,
            beforeRequest: function() {
                setLoadingState(true);
            },
            afterRequest: function() {
                setLoadingState(false);
            }
        });
    }
    
    // Validate form data
    function validateForm(data) {
        const errors = [];
        
        if (!data.identity || data.identity.trim() === '') {
            errors.push('Email or username is required');
        }
        
        if (!data.password || data.password.trim() === '') {
            errors.push('Password is required');
        }
        
        if (errors.length > 0) {
            showErrors(errors);
            return false;
        }
        
        hideErrors();
        return true;
    }
    
    // Show error messages
    function showErrors(errors) {
        if (formErrors && errors.length > 0) {
            const errorMessage = formErrors.querySelector('.error-message');
            if (errorMessage) {
                errorMessage.textContent = errors.join(', ');
                formErrors.style.display = 'block';
            }
        }
    }
    
    // Hide error messages
    function hideErrors() {
        if (formErrors) {
            formErrors.style.display = 'none';
        }
    }
    
    // Set loading state
    function setLoadingState(loading) {
        if (submitButton) {
            if (loading) {
                submitButton.classList.add('loading');
                submitButton.disabled = true;
            } else {
                submitButton.classList.remove('loading');
                submitButton.disabled = false;
            }
        }
    }
    
    // Clear form on successful submission
    function clearForm() {
        if (loginForm) {
            loginForm.reset();
            hideErrors();
        }
    }
    
    // Handle successful login (redirect)
    function handleSuccessfulLogin(response) {
        // Check if response indicates success
        if (response.success) {
            // Redirect to dashboard or home page
            window.location.href = response.redirectUrl || '/';
        }
    }
    
    // Add input event listeners for real-time validation
    const inputs = loginForm?.querySelectorAll('.form-input');
    if (inputs) {
        inputs.forEach(input => {
            input.addEventListener('input', function() {
                // Clear errors when user starts typing
                if (formErrors && formErrors.style.display !== 'none') {
                    hideErrors();
                }
            });
        });
    }

    function handleFormSubmit(event) {
        event.preventDefault();
        
        // Show loading state
        setLoadingState(true);
        
        // Get form data
        const formData = new FormData(loginForm);
        const data = Object.fromEntries(formData);
        
        // Basic validation
        if (!validateForm(data)) {
            setLoadingState(false);
            return;
        }
        
        // Submit form via HTMX
        htmx.ajax('POST', '/api/v1/auth/login', {
            target: '#login-form',
            swap: 'outerHTML',
            values: data,
            beforeRequest: function() {
                setLoadingState(true);
            },
            afterRequest: function() {
                setLoadingState(false);
            }
        });
    }
    
    // Validate form data
    function validateForm(data) {
        const errors = [];
        
        if (!data.identity || data.identity.trim() === '') {
            errors.push('Email or username is required');
        }
        
        if (!data.password || data.password.trim() === '') {
            errors.push('Password is required');
        }
        
        if (errors.length > 0) {
            showErrors(errors);
            return false;
        }
        
        hideErrors();
        return true;
    }
    
    // Show error messages
    function showErrors(errors) {
        if (formErrors && errors.length > 0) {
            const errorMessage = formErrors.querySelector('.error-message');
            if (errorMessage) {
                errorMessage.textContent = errors.join(', ');
                formErrors.style.display = 'block';
            }
        }
    }
    
    // Hide error messages
    function hideErrors() {
        if (formErrors) {
            formErrors.style.display = 'none';
        }
    }
    
    // Set loading state
    function setLoadingState(loading) {
        if (submitButton) {
            if (loading) {
                submitButton.classList.add('loading');
                submitButton.disabled = true;
            } else {
                submitButton.classList.remove('loading');
                submitButton.disabled = false;
            }
        }
    }
    
    // Clear form on successful submission
    function clearForm() {
        if (loginForm) {
            loginForm.reset();
            hideErrors();
        }
    }
    
    // Handle successful login (redirect)
    function handleSuccessfulLogin(response) {
        // Check if response indicates success
        if (response.success) {
            // Redirect to dashboard or home page
            window.location.href = response.redirectUrl || '/';
        }
    }
    
    // Add input event listeners for real-time validation
    const inputs = loginForm?.querySelectorAll('.form-input');
    if (inputs) {
        inputs.forEach(input => {
            input.addEventListener('input', function() {
                // Clear errors when user starts typing
                if (formErrors && formErrors.style.display !== 'none') {
                    hideErrors();
                }
            });
        });
    }
});

document.addEventListener('DOMContentLoaded', function() {
    const loginForm = document.getElementById('login-form');
    const submitButton = document.getElementById('login-submit');
    const formErrors = document.getElementById('form-errors');
    
    if (loginForm) {
        loginForm.addEventListener('submit', handleFormSubmit);
    }
    
    // Handle form submission
    function handleFormSubmit(event) {
        event.preventDefault();
        
        // Show loading state
        setLoadingState(true);
        
        // Get form data
        const formData = new FormData(loginForm);
        const data = Object.fromEntries(formData);
        
        // Basic validation
        if (!validateForm(data)) {
            setLoadingState(false);
            return;
        }
        
        // Submit form via HTMX
        htmx.ajax('POST', '/api/v1/auth/login', {
            target: '#login-form',
            swap: 'outerHTML',
            values: data,
            beforeRequest: function() {
                setLoadingState(true);
            },
            afterRequest: function() {
                setLoadingState(false);
            }
        });
    }
    
    // Validate form data
    function validateForm(data) {
        const errors = [];
        
        if (!data.identity || data.identity.trim() === '') {
            errors.push('Email or username is required');
        }
        
        if (!data.password || data.password.trim() === '') {
            errors.push('Password is required');
        }
        
        if (errors.length > 0) {
            showErrors(errors);
            return false;
        }
        
        hideErrors();
        return true;
    }
    
    // Show error messages
    function showErrors(errors) {
        if (formErrors && errors.length > 0) {
            const errorMessage = formErrors.querySelector('.error-message');
            if (errorMessage) {
                errorMessage.textContent = errors.join(', ');
                formErrors.style.display = 'block';
            }
        }
    }
    
    // Hide error messages
    function hideErrors() {
        if (formErrors) {
            formErrors.style.display = 'none';
        }
    }
    
    // Set loading state
    function setLoadingState(loading) {
        if (submitButton) {
            if (loading) {
                submitButton.classList.add('loading');
                submitButton.disabled = true;
            } else {
                submitButton.classList.remove('loading');
                submitButton.disabled = false;
            }
        }
    }
    
    // Clear form on successful submission
    function clearForm() {
        if (loginForm) {
            loginForm.reset();
            hideErrors();
        }
    }
    
    // Handle successful login (redirect)
    function handleSuccessfulLogin(response) {
        // Check if response indicates success
        if (response.success) {
            // Redirect to dashboard or home page
            window.location.href = response.redirectUrl || '/';
        }
    }

            input.addEventListener('input', function() {
                // Clear errors when user starts typing
                if (formErrors && formErrors.style.display !== 'none') {

                }
            });
        });
